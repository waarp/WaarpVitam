40  /**
41   * Class for Waarp sending back to Waarp Partner from Monitor using native R66
42   */
43  class ManagerToWaarpR66 implements ManagerToWaarp {
44    /**
45     * Internal Logger
46     */
47    private static final WaarpLogger logger =
48        WaarpLoggerFactory.getLogger(ManagerToWaarpR66.class);
49  
50    ManagerToWaarpR66() {
51      // nothing
52    }
53  
54    @Override
55    public boolean sendBackInformation(
56        final IngestRequestFactory ingestRequestFactory,
57        final IngestRequest ingestRequest, final String filename,
58        final String fileInfo) throws InvalidParseOperationException {
59      logger.debug("Will send {} while step is {}", filename,
60                   ingestRequest.getStep());
61      R66Future future = new R66Future(true);
62      SubmitTransfer submitTransfer =
63          new SubmitTransfer(future, ingestRequest.getWaarpPartner(), filename,
64                             ingestRequest.getWaarpRule(),
65                             ingestRequest.getRequestId() + ' ' + fileInfo, true,
66                             Configuration.configuration.getBlockSize(),
67                             DbConstantR66.ILLEGALVALUE, null);
68      submitTransfer.run();
69      future.awaitOrInterruptible();
70      if (future.isSuccess()) {
71        ingestRequest.setWaarpId(future.getResult().getRunner().getSpecialId());
72        ingestRequest.save(ingestRequestFactory);
73        return waitForAllDone(ingestRequest);
74      }
75      return false;
76    }
77  
78    @Override
79    public boolean sendBackInformation(final DipRequestFactory dipRequestFactory,
80                                       final DipRequest dipRequest,
81                                       final String filename,
82                                       final String fileInfo)
83        throws InvalidParseOperationException {
84      logger
85          .debug("Will send {} while step is {}", filename, dipRequest.getStep());
86      R66Future future = new R66Future(true);
87      SubmitTransfer submitTransfer =
88          new SubmitTransfer(future, dipRequest.getWaarpPartner(), filename,
89                             dipRequest.getWaarpRule(),
90                             dipRequest.getRequestId() + ' ' + fileInfo, true,
91                             Configuration.configuration.getBlockSize(),
92                             DbConstantR66.ILLEGALVALUE, null);
93      submitTransfer.run();
94      future.awaitOrInterruptible();
95      if (future.isSuccess()) {
96        dipRequest.setWaarpId(future.getResult().getRunner().getSpecialId());
97        dipRequest.save(dipRequestFactory);
98        return waitForAllDone(dipRequest);
99      }
100     return false;
101   }
102 
103   /**
104    * Ensure that SubmitTransfer is done totally (file sent) before continuing
105    *
106    * @param abstractVitamRequest
107    *
108    * @return True if done
109    */
110   private boolean waitForAllDone(AbstractVitamRequest abstractVitamRequest) {
111     while (true) {
112       try {
113         DbHostAuth dbHostAuth =
114             new DbHostAuth(abstractVitamRequest.getWaarpPartner());
115         DbTaskRunner checkedRunner =
116             new DbTaskRunner(abstractVitamRequest.getWaarpId(),
117                              Configuration.configuration
118                                  .getHostId(dbHostAuth.isSsl()),
119                              abstractVitamRequest.getWaarpPartner());
120         if (checkedRunner.isAllDone()) {
121           logger.info("DbTaskRunner done");
122           return true;
123         } else if (checkedRunner.isInError()) {
124           logger.warn("DbTaskRunner in error for {}", abstractVitamRequest);
125           return false;
126         }
127         Thread.sleep(500);
128       } catch (InterruptedException e) {//NOSONAR
129         logger.error("Interrupted", e);
130         return false;
131       } catch (WaarpDatabaseException e) {
132         logger.error("Cannot found DbTaskRunner", e);
133         return false;
134       } catch (OpenR66ProtocolNoSslException e) {
135         logger.error("Cannot found HostSslId", e);
136         return false;
137       }
138     }
139   }
